class BitPackedMaskManager:
    def __init__(self, d_in: int, d_sae: int):
        self.d_in = d_in
        self.d_sae = d_sae
        self.masks = torch.ones(d_in, d_sae, dtype=torch.bool)
        self.correlation_cache = {}
        
    def update_masks(self, activations: torch.Tensor, threshold: float = 0.5):
        # Calculate correlations if not in cache
        batch_key = str(activations.shape)
        if batch_key not in self.correlation_cache:
            corr_matrix = F.normalize(activations.T) @ F.normalize(activations)
            self.correlation_cache[batch_key] = corr_matrix
        
        # Update masks based on correlations
        corr_matrix = self.correlation_cache[batch_key]
        self.masks = (corr_matrix < threshold)
        return self.masks

    def apply_masks(self, weights: torch.Tensor) -> torch.Tensor:
        return weights * self.masks.to(weights.device)
