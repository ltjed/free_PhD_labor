    def loss(self, x, logging=False, **kwargs):
        x_hat, f = self.ae(x, output_features=True)
        
        # Adversarial component with stop gradient
        with torch.no_grad():
            h_spurious = f @ self.ae.W_adv
        adv_loss = torch.relu(self.ae.margin - (f * h_spurious).sum(dim=-1)).mean()
        
        # Update EMA statistics
        sim = (f * h_spurious).sum(dim=-1)
        self.ae.ema_sim.data = 0.95 * self.ae.ema_sim + 0.05 * sim.mean(dim=0)
        self.ae.ema_std.data = 0.95 * self.ae.ema_std + 0.05 * sim.std(dim=0)
        self.ae.margin.data = self.ae.ema_sim + 2 * self.ae.ema_std
