    def get_top_pairs(self, features):
        """Select top-k feature pairs based on activation correlation."""
        B = features.size(0)  # Batch size
        F = features.size(1)  # Number of features
        
        # Compute pairwise feature correlations
        features_norm = features / (features.norm(dim=0, keepdim=True) + 1e-8)
        correlations = torch.abs(features_norm.T @ features_norm / B)
        
        # Get top-k pairs (excluding self-correlations)
        mask = torch.ones_like(correlations, dtype=torch.bool)
        mask.fill_diagonal_(0)
        correlations = correlations * mask
        
        k = int(self.topk_percent * F * (F-1) / 2)
        top_corr, indices = torch.topk(correlations.view(-1), k)
        
        # Convert flat indices to 2D indices
        i, j = indices // F, indices % F
        return i, j, top_corr

    def orthogonality_loss(self, features):
        """Compute orthogonality loss for selected feature pairs."""
        i, j, _ = self.get_top_pairs(features)
        
        # Normalize features
        features_norm = features / (features.norm(dim=0, keepdim=True) + 1e-8)
        
        # Compute dot products for selected pairs
        dot_products = torch.sum(features_norm[:, i] * features_norm[:, j], dim=0)
        
        # Orthogonality loss
        loss = torch.mean(dot_products.pow(2))
        return loss
